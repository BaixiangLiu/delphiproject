VAR
   TF :TEXTFILE;
   OUT_PORT :STRING;


BEGIN
   AssignFile(TF,OUT_PORT);
   REWRITE(TF);
   WRITELN(TF,'SIZE '+C_WIDTH+','+C_HEIGHT);
   WRITELN(TF,'CLS');
   WRITELN(TF,'SPEED '+C_SPEED);
   WRITELN(TF,'DENSITY '+C_DENSITY);
   WRITELN(TF,'SET CUTTER OFF');
   WRITELN(TF,'DIRECTION '+C_DIR);
   WRITELN(TF,'REFERENCE '+C_REF1+','+C_REF2);
   WRITELN(TF,'GAP '+C_GAP1+','+C_GAP2);
   CloseFile(TF);

END;
